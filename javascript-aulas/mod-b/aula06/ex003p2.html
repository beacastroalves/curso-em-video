Como transformar String para Number e String para Number

  • // (number + number) usa o + para adição
  • // (string + string) usa o + para concatenação
  •         
  Number.parseInt(n) // Para numeros inteiros
  Number.parseFloat(n) // Para numeros reais (com virgula)
  Number(n)

  n.toString() // Para converter um número em string
  String(n) // Para converter um número em string

  var n1 = Number(window.prompt('Digite um número: '));
  var n2 = Number(window.prompt('Digite outro número: '));
  var s = n1 + n2;
  window.alert('A soma dos valores é ' + s);

  var j = 'Javascript';
  'Eu estou aprendendo j' // Não faz interpolação
  'Eu estou aprendendo ' + j // usa concatenação
  `Eu estou aprendendo ${j}` // usa template string

  var j = 'Javascript';
  j.length // quantos caracteres a string tem
  j.toUpperCase() // Transforma tudo em 'MAIÚSCULAS'
  j.toLowerCase() // Transforma tudo em 'minúsculas'

  <script>
    var n1 = Number(window.prompt('Digite um número: '));
