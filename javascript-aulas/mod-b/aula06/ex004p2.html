// toFixed(2) é para depois do ponto em um numero real ele apresentar duas casas decimais

    
  var n1 = 1545.5;
  n1;
  n1.toFixed(2);
  > 1545.50
  n1.toFixed(2).replace('.', ',');
  > 1545,50
  n1.toLocaleString('pt-BR'), {'style: 'currency', currency: 'BRL'});
  > 'R$ 1,545.50'
  document.write(n1.toLocaleString('pt-BR'), {'style: 'currency', currency: 'USD'});
  > 'US$ 1,545.50'
  document.write(n1.toLocaleString('pt-BR'), {'style: 'currency', currency: 'EUR'});
  > '€ 1,545.50'
